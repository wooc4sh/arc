% --- OTWARCIE KONTEKSTU (Styl Symetryczny) ---
\newcommand{\contextsection}[1]{%
    \noindent
        % Kolor i czcionka dla całości
        \color{Black}
        % Lewa linia

        \noindent
        \raisebox{0ex}{\small $\blacksquare$} 
	\sffamily\bfseries #1
	\rule[0.5ex]{0.8\linewidth}{1pt} 

    \par\vspace{1.5em} % Odstęp po tytule przed treścią
    \noindent\ignorespaces % Usuwa wcięcie pierwszego akapitu
}

% --- ZAMKNIĘCIE KONTEKSTU ---
\newcommand{\contextend}{%
    \par\vspace{2em}
    \noindent\begin{center}
        \color{Orchid}
        % Linia - Kropka - Linia
        \rule[0.5ex]{0.3\linewidth}{0.5pt} 
        \hspace{0.5em}
        \raisebox{0ex}{\scriptsize $\blacksquare$} 
        \hspace{0.5em}
        \rule[0.5ex]{0.3\linewidth}{0.5pt}
    \end{center}
    \par\vspace{3em} % Odstęp po zamknięciu
}
