\begin{problem}[Zadanie 2.]

        let y = 10

\end{problem}

\begin{solution}
Rozwiązanie polega na zdefiniowaniu funkcji:
\begin{lstlisting}[style=ocamlStyle]

let f y = x + y
\end{lstlisting}
co kończy zadanie.
\end{solution}
